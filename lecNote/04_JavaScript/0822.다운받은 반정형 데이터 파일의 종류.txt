    location,min,max

    '부산',26,31

     [{'location':'서울', 'min':24, 'max':30},
     {'location':'부산', 'min':26, 'max':31}]

	<location>서울</location>
	<min>24</min>
	<max>30</max>
    </section>
     <section>
	<location>부산</location>
	<min>26</min>

    </section>
